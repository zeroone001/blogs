#### instanceof
```javascript
function instanceofFun (left, right) {
    left = left._proto_;
    var prototype = right.prototype;
    while(true) {
        if (left === null || left === undefined) {
            return false;
        }
        if (prototype === left) {
            return true;
        }
        left = left._proto_;
    }
}
```
